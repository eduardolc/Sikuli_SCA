def delta(dataInicio,dataFim):
    dataDelta = time.strftime("%M:%S",time.localtime(dataFim - dataInicio))
    return dataDelta

def dropDown(qtdDropDown, tmpWait):
    cont = 0
    while (cont <= qtdDropDown):
        type(Key.DOWN)
        wait(tmpWait)
        cont = cont + 1

def tabOrder(qtdTabOrder, tmpWait):
    cont = 1
    while (cont <= qtdTabOrder):
        type(Key.TAB)
        wait(tmpWait)
        cont = cont + 1
